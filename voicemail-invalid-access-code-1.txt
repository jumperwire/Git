<?xml version="1.0" encoding="UTF-8"?>
<vxml version="2.1">
	<property name="inputmodes" value="dtmf"/>
	<form id="get_response">
		<block>
			<prompt xml:lang="en-gb-male">
				<prosody rate="-10%">
					Welcome to NCS Voicemail,
				</prosody>
			</prompt>
		</block>
		<field name="response" type="digits?minlength=1;maxlength=10">
			<grammar mode="dtmf" version="1.0" root="root">
				<rule id="root" scope="public">
					<one-of>
						<item> 1 </item>
						<item> 2 </item>
						<item> 3 </item>
						<item> 4 </item>
						<item> 5 </item>
						<item> 6 </item>
						<item> 7 </item>
						<item> 8 </item>
						<item> 9 </item>
						<item> 0 </item>
						<item> # </item>
					</one-of>
				</rule>
			</grammar>
			<prompt xml:lang="en-gb-male">
				<prosody rate="-10%">
					To listen to the broadcast message, please enter the access code followed by the # sign.
				</prosody>
			</prompt>
			<noinput>
				<prompt xml:lang="en-gb-male">
					<prosody rate="-10%">
						You have not entered any response, please try again.
					</prosody>
				</prompt>
				<reprompt/>
			</noinput>
			<nomatch>
				<prompt xml:lang="en-gb-male">
					<prosody rate="-10%">
						You have entered invalid response, please try again.
					</prosody>
				</prompt>
				<reprompt/>
			</nomatch>
			<filled>
				<prompt xml:lang="en-gb-male">
					<prosody rate="-10%">
						You have entered, <prosody rate="-50%"><say-as interpret-as="vxml:digits"><value expr="response"/></say-as></prosody>.
					</prosody>
				</prompt>
				<if cond="response == '0'">
					<goto next="#get_response"/>
				</if>
				<else/>
				<submit next="https://raw.githubusercontent.com/jumperwire/Git/master/voicemail-invalid-access-code-2.txt" method="get" namelist="response"/>
			</filled>
		</field>
	</form>
</vxml>
